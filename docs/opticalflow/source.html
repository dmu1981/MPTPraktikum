  8# If fewer than this many points are left, we detect a fresh set of corners.
  9# This keeps the demo useful even when the tracked object leaves the image.
 10MIN_TRACKS = 15
 11
 12# Parameters for cv2.goodFeaturesToTrack. KLT works best on image patches that
 13# contain visible structure in both directions, therefore we track corners.
 14FEATURE_PARAMS = dict(
 15    # Track at most this many points. A small number keeps the visualization
 16    # readable and makes the algorithm fast enough for a webcam demo.
 17    maxCorners=100,
 18    # Keep points whose corner quality is at least 1 percent of the best
 19    # detected corner. Raising this value keeps only very strong corners.
 20    qualityLevel=0.01,
 21    # Keep detected points separated by at least this many pixels. Without this,
 22    # many points can cluster on the same textured area.
 23    minDistance=10,
 24    # Size of the local neighborhood used for the corner score.
 25    blockSize=7,
 26)
 27
 28# Parameters for cv2.calcOpticalFlowPyrLK. OpenCV implements the pyramidal
 29# Lucas-Kanade tracker, which can follow motions larger than a few pixels.
 30LK_PARAMS = dict(
 31    # The tracker compares a small window around each point. A larger window is
 32    # more stable, but also assumes the same motion over a larger image region.
 33    winSize=(21, 21),
 34    # Number of pyramid levels. 0 means only the original image. 3 means the
 35    # original image plus three smaller versions.
 36    maxLevel=3,
 37    # Stop after at most 30 iterations, or earlier when the point update is
 38    # smaller than epsilon. This is the standard OpenCV stopping criterion.
 39    criteria=(
 40        cv2.TERM_CRITERIA_EPS | cv2.TERM_CRITERIA_COUNT,
 41        30,
 42        0.01,
 43    ),
 44)
 45
 46# Keyboard controls used in the main loop.
 47RESET_KEY = ord("r")
 48QUIT_KEYS = {27, ord("q")}  # 27 is the ESC key.

 51def toGray(frame):
 52    """
 53    Convert one camera frame to a grayscale image.
 54
 55    The camera frame delivered by OpenCV has three color channels in BGR order.
 56    Both goodFeaturesToTrack and calcOpticalFlowPyrLK operate on single-channel
 57    images, therefore we use a grayscale version for the actual computation.
 58
 59    Args:
 60        frame (np.ndarray): BGR camera image.
 61
 62    Returns:
 63        np.ndarray: Grayscale image with the same width and height.
 64    """
 65    return cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)

 68def detectFeaturePoints(gray):
 69    """
 70    Detect good points that can be tracked by KLT.
 71
 72    We use Shi-Tomasi corners via cv2.goodFeaturesToTrack. These are exactly
 73    the kind of points where the 2x2 KLT equation system is well-conditioned:
 74    the image patch changes clearly in both x- and y-direction.
 75
 76    Args:
 77        gray (np.ndarray): Current grayscale image.
 78
 79    Returns:
 80        np.ndarray | None: Points with shape (N, 1, 2), or None if no useful
 81        corners were found.
 82    """
 83    return cv2.goodFeaturesToTrack(gray, **FEATURE_PARAMS)
 84
 85
 86def trackFeaturePoints(previousGray, gray, previousPoints):
 87    """
 88    Track points from the previous frame into the current frame.
 89
 90    cv2.calcOpticalFlowPyrLK receives the previous image, the current image and
 91    the points in the previous image. For each point it estimates where the same
 92    image patch moved in the current image.
 93
 94    Args:
 95        previousGray (np.ndarray): Grayscale image from the last time step.
 96        gray (np.ndarray): Grayscale image from the current time step.
 97        previousPoints (np.ndarray | None): Points from the previous image with
 98        shape (N, 1, 2).
 99
100    Returns:
101        tuple: (currentPoints, status, error) as returned by OpenCV. If there
102        are no points to track, all three entries are None.
103    """
104    if previousPoints is None or len(previousPoints) == 0:
105        return None, None, None
106
107    return cv2.calcOpticalFlowPyrLK(
108        previousGray,
109        gray,
110        previousPoints,
111        None,
112        **LK_PARAMS,
113    )

116def keepGoodTracks(previousPoints, currentPoints, status):
117    """
118    Keep only tracks that OpenCV marked as successful.
119
120    The Lucas-Kanade tracker returns a status value for every input point.
121    status == 1 means a plausible corresponding point was found. status == 0
122    means tracking failed, for example because the point left the image or the
123    local patch no longer matches.
124
125    Args:
126        previousPoints (np.ndarray): Points in the previous image.
127        currentPoints (np.ndarray | None): Tracked points in the current image.
128        status (np.ndarray | None): Success flag for every point.
129
130    Returns:
131        tuple[np.ndarray, np.ndarray]: Matching arrays with shape (M, 2):
132        successful previous points and their successful current positions.
133    """
134    if currentPoints is None or status is None:
135        empty = np.empty((0, 2), dtype=np.float32)
136        return empty, empty
137
138    good = status.reshape(-1).astype(bool)
139    goodPrevious = previousPoints[good].reshape(-1, 2)
140    goodCurrent = currentPoints[good].reshape(-1, 2)
141
142    return goodPrevious, goodCurrent

145def drawTracks(frame, tracks, previousPoints, currentPoints):
146    """
147    Draw the optical flow vectors on top of the live camera image.
148
149    The variable tracks is a persistent image. Every new motion segment is drawn
150    into this image, so older paths remain visible. The current feature
151    positions are drawn onto a copy of the current camera frame.
152
153    Args:
154        frame (np.ndarray): Current BGR camera image.
155        tracks (np.ndarray): Persistent BGR image containing old line segments.
156        previousPoints (np.ndarray): Successful point locations in the previous
157        frame with shape (M, 2).
158        currentPoints (np.ndarray): Successful point locations in the current
159        frame with shape (M, 2).
160
161    Returns:
162        np.ndarray: Camera image with accumulated flow lines and current points.
163    """
164    overlay = frame.copy()
165
166    for previousPoint, currentPoint in zip(previousPoints, currentPoints):
167        oldX, oldY = previousPoint.ravel()
168        newX, newY = currentPoint.ravel()
169
170        oldPosition = (int(oldX), int(oldY))
171        newPosition = (int(newX), int(newY))
172
173        # Green lines show where a point came from.
174        cv2.line(tracks, oldPosition, newPosition, (0, 255, 0), 2)
175
176        # Red circles show the current point positions.
177        cv2.circle(overlay, newPosition, 4, (0, 0, 255), -1)
178
179    # Add the persistent track image to the current frame.
180    return cv2.add(overlay, tracks)

183def drawStatus(frame, trackCount):
184    """
185    Draw a small status text into the output image.
186
187    This is not part of the algorithm. It only helps during experimentation:
188    students can immediately see how many points are still being tracked.
189
190    Args:
191        frame (np.ndarray): Output image that will be modified in-place.
192        trackCount (int): Number of currently successful tracks.
193    """
194    text = f"tracks: {trackCount}   r: reset   ESC/q: quit"
195    cv2.putText(
196        frame,
197        text,
198        (12, 28),
199        cv2.FONT_HERSHEY_SIMPLEX,
200        0.7,
201        (255, 255, 255),
202        2,
203        cv2.LINE_AA,
204    )

207def mainLoop():
208    """
209    Open the webcam and run the KLT tracker until the user quits.
210    """
211    cam = cv2.VideoCapture(0)
212    if not cam.isOpened():
213        print("Cannot open camera")
214        return
215
216    # Read one frame before the loop. KLT always compares two neighboring
217    # images, therefore we need an initial "previous" image.
218    ret, frame = cam.read()
219    if not ret:
220        print("Cannot read first camera image")
221        cam.release()
222        return
223
224    previousGray = toGray(frame)
225    previousPoints = detectFeaturePoints(previousGray)
226
227    # The track image has the same size as the camera frame. It starts black and
228    # stores the accumulated green motion lines.
229    tracks = np.zeros_like(frame)
230
231    while True:
232        ret, frame = cam.read()
233        if not ret:
234            break
235
236        gray = toGray(frame)
237
238        currentPoints, status, _ = trackFeaturePoints(
239            previousGray,
240            gray,
241            previousPoints,
242        )
243        goodPrevious, goodCurrent = keepGoodTracks(
244            previousPoints,
245            currentPoints,
246            status,
247        )
248
249        output = drawTracks(frame, tracks, goodPrevious, goodCurrent)
250        drawStatus(output, len(goodCurrent))
251        cv2.imshow(WINDOW_NAME, output)
252
253        key = cv2.waitKey(1) & 0xFF
254        if key in QUIT_KEYS:
255            break
256
257        # Redetect points when the user asks for it or when too few tracks are
258        # left. We also clear the path image so old and new tracks do not mix.
259        if key == RESET_KEY or len(goodCurrent) < MIN_TRACKS:
260            tracks = np.zeros_like(frame)
261            previousGray = gray
262            previousPoints = detectFeaturePoints(previousGray)
263            continue
264
265        # The current frame becomes the previous frame for the next iteration.
266        # OpenCV expects the points again in shape (N, 1, 2).
267        previousGray = gray.copy()
268        previousPoints = goodCurrent.reshape(-1, 1, 2)
269
270    cam.release()
271    cv2.destroyAllWindows()
272
273
274if __name__ == "__main__":
275    mainLoop()
